## InvoicesTable

Displays a paginated table of invoices with customer information, amounts, dates, and status. Includes responsive layouts for mobile and desktop views with edit and delete actions.

<ParamField path="query" type="string" required>
  Search query string for filtering invoices
</ParamField>

<ParamField path="currentPage" type="number" required>
  Current page number for pagination
</ParamField>

### Usage

```tsx
import InvoicesTable from '@/app/ui/invoices/table';

export default async function InvoicesPage({ 
  searchParams 
}: { 
  searchParams?: { query?: string; page?: string } 
}) {
  const query = searchParams?.query || '';
  const currentPage = Number(searchParams?.page) || 1;
  
  return (
    <InvoicesTable query={query} currentPage={currentPage} />
  );
}
```

## Pagination

Client-side pagination component with page numbers and arrow navigation. Automatically generates page numbers with ellipsis for large page counts.

<ParamField path="totalPages" type="number" required>
  Total number of pages available
</ParamField>

### Usage

```tsx
import Pagination from '@/app/ui/invoices/pagination';
import { fetchInvoicesPages } from '@/app/lib/data';

export default async function InvoicesPage({ 
  searchParams 
}: { 
  searchParams?: { query?: string } 
}) {
  const query = searchParams?.query || '';
  const totalPages = await fetchInvoicesPages(query);
  
  return (
    <div className="mt-5 flex w-full justify-center">
      <Pagination totalPages={totalPages} />
    </div>
  );
}
```

### Internal components

The `Pagination` component includes two helper components:

- **PaginationNumber** - Individual page number button
- **PaginationArrow** - Previous/next arrow buttons

## Form

Form component for creating new invoices. Includes customer selection, amount input, and status radio buttons.

<ParamField path="customers" type="CustomerField[]" required>
  Array of available customers to select from
</ParamField>

### CustomerField type

```typescript
type CustomerField = {
  id: string;
  name: string;
};
```

### Usage

```tsx
import Form from '@/app/ui/invoices/create-form';
import { fetchCustomers } from '@/app/lib/data';

export default async function CreateInvoice() {
  const customers = await fetchCustomers();
  
  return (
    <main>
      <Breadcrumbs
        breadcrumbs={[
          { label: 'Invoices', href: '/dashboard/invoices' },
          { label: 'Create Invoice', href: '/dashboard/invoices/create', active: true },
        ]}
      />
      <Form customers={customers} />
    </main>
  );
}
```

## EditInvoiceForm

Form component for editing existing invoices. Similar to the create form but pre-populated with invoice data.

<ParamField path="invoice" type="InvoiceForm" required>
  The invoice object to edit
</ParamField>

<ParamField path="customers" type="CustomerField[]" required>
  Array of available customers to select from
</ParamField>

### InvoiceForm type

```typescript
type InvoiceForm = {
  id: string;
  customer_id: string;
  amount: number;
  status: 'pending' | 'paid';
};
```

### Usage

```tsx
import EditInvoiceForm from '@/app/ui/invoices/edit-form';
import { fetchInvoiceById, fetchCustomers } from '@/app/lib/data';

export default async function EditInvoice({ params }: { params: { id: string } }) {
  const id = params.id;
  const [invoice, customers] = await Promise.all([
    fetchInvoiceById(id),
    fetchCustomers(),
  ]);
  
  return (
    <main>
      <Breadcrumbs
        breadcrumbs={[
          { label: 'Invoices', href: '/dashboard/invoices' },
          { label: 'Edit Invoice', href: `/dashboard/invoices/${id}/edit`, active: true },
        ]}
      />
      <EditInvoiceForm invoice={invoice} customers={customers} />
    </main>
  );
}
```

## Breadcrumbs

Breadcrumb navigation component that shows the current page hierarchy.

<ParamField path="breadcrumbs" type="Breadcrumb[]" required>
  Array of breadcrumb items to display
</ParamField>

### Breadcrumb type

```typescript
interface Breadcrumb {
  label: string;
  href: string;
  active?: boolean;
}
```

### Usage

```tsx

```
